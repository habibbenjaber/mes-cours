function CounterList() {
  const [counters, setCounters] = React.useState([0, 0, 0]);

    setCounters(prev => prev.map((count, i) => i === index ? newValue : count));
  };

  const resetAll = () => {
    setCounters([0, 0, 0]);
  };

  return (
    <div>
      <div style={{ marginBottom: '1rem' }}>
        <button onClick={resetAll}>RÃ©initialiser tous les compteurs</button>
      </div>
      {counters.map((count, index) => (
        <Counter
          key={index}
          count={count}
          onIncrement={() => updateCounter(index, count + 1)}
          onDecrement={() => updateCounter(index, count - 1)}
          onReset={() => updateCounter(index, 0)}

      <div style={{ marginTop: '1rem', padding: '1rem', backgroundColor: '#f8f9fa' }}>
