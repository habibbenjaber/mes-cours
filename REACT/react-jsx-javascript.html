function ComplexExpressions() {
  const products = [
    { id: 1, name: "Laptop", price: 999, category: "electronics" },
    { id: 2, name: "Book", price: 25, category: "education" },
    { id: 3, name: "Headphones", price: 150, category: "electronics" },
    { id: 4, name: "Notebook", price: 5, category: "education" }
  ];

  const [filter, setFilter] = React.useState('all');

  const filteredProducts = products.filter(product =>
    filter === 'all' || product.category === filter
  );

  const totalValue = filteredProducts.reduce(
    (sum, product) => sum + product.price, 0
  );

  const averagePrice = filteredProducts.length > 0
    ? totalValue / filteredProducts.length
    : 0;

  return (
    <div>
      <div className="filters">
        {['all', 'electronics', 'education'].map(category => (
          <button
            key={category}
            onClick={() => setFilter(category)}
            style={{
              background: filter === category ? '#61dafb' : '#f8f9fa'
            }}
          >
            {category === 'all' ? 'Tous' : category}
          </button>
        ))}
      </div>
      
      <div className="stats">
        <p>Nombre de produits: {filteredProducts.length}</p>
        <p>Valeur totale: {totalValue} dt </p>
        <p>Prix moyen: {averagePrice.toFixed(2)} dt </p>
      </div>
      
      <div className="products">
        {filteredProducts.map(product => (
          <div key={product.id} className="product-card">
            <h4>{product.name}</h4>
            <p>Prix: {product.price} dt</p>
            <p>Cat√©gorie: {product.category}</p>
            <p style={{
              color: product.price > averagePrice ? '#dc3545' : '#28a745',
              fontWeight: 'bold'
            }}>
              {product.price > averagePrice ? 'Au-dessus' : 'En-dessous'} de la moyenne
            </p>
          </div>
        ))}
      </div>
    </div>
  );
}
